_CUTOFF = 1024
def kara(x, y):
    if x.bit_length() <= _CUTOFF or y.bit_length() <= _CUTOFF:  # Base case
        #print('naive')
        return naive(x, y)
    else:
        #print('karatsuba')
        n = max(x.bit_length(), y.bit_length())
        half = (n + 32) // 64 * 32
        mask = (1 << half) - 1
        xlow = x & mask
        ylow = y & mask
        xhigh = x >> half
        yhigh = y >> half
        a = kara(xhigh, yhigh)
        b = kara(xlow + xhigh, ylow + yhigh)
        c = kara(xlow, ylow)
        d = b - a - c
        return (((a << half) + d) << half) + c
